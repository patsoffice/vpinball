# plugin CMake file to be included in main project CMakelists

# --- ChannelVol plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(CHANNELVOL_PLUGIN_SOURCES
   plugins/channelvol/channelvol.cpp
   plugins/channelvol/CoreAudioMixer.cpp
)

if(BUILD_SHARED)
   add_library(ChannelVolPlugin MODULE ${CHANNELVOL_PLUGIN_SOURCES})

   target_include_directories(ChannelVolPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/channelvol
   )

   if(PluginPlatform STREQUAL "windows" OR PluginPlatform STREQUAL "windows-mingw")
      if(MSVC)
         add_compile_options(
            $<$<CONFIG:RELEASE>:/Ob2>
            $<$<CONFIG:RELEASE>:/O2>
            $<$<CONFIG:RELEASE>:/Oi>
            $<$<CONFIG:RELEASE>:/arch:SSE2>
            $<$<CONFIG:RELEASE>:/fp:fast>
            $<$<CONFIG:RELEASE>:/fp:except->
            $<$<CONFIG:RELEASE>:/Ot>
            $<$<CONFIG:RELEASE>:/GF>
            $<$<CONFIG:RELEASE>:/GS->
            $<$<CONFIG:RELEASE>:/Gy>
            $<$<CONFIG:RELEASE>:/GR>
            $<$<CONFIG:RELEASE>:/Oy>
            $<$<CONFIG:RELEASE>:/GT>
            $<$<CONFIG:RELEASE>:/GL>
         )
         set_target_properties(ChannelVolPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
         )
         target_compile_options(ChannelVolPlugin PUBLIC
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(ChannelVolPlugin PUBLIC
            $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
            $<$<CONFIG:RELEASE>:/OPT:REF>
            $<$<CONFIG:RELEASE>:/OPT:ICF>
            $<$<CONFIG:RELEASE>:/LTCG>
         )
      endif()
      if (PluginArch STREQUAL "x64")
         set_target_properties(ChannelVolPlugin PROPERTIES
            PREFIX ""
            OUTPUT_NAME "plugin-channelvol64"
         )
      else()
         set_target_properties(ChannelVolPlugin PROPERTIES
            PREFIX ""
            OUTPUT_NAME "plugin-channelvol"
         )
      endif()
   else()
      set_target_properties(ChannelVolPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-channelvol"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(ChannelVolPlugin PROPERTIES
            SUFFIX ".dylib"
         )
         target_link_libraries(ChannelVolPlugin PRIVATE
            "-framework CoreAudio"
            "-framework AudioToolbox"
            "-framework CoreFoundation"
         )
      endif()
   endif()

   add_custom_command(TARGET ChannelVolPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/channelvol/plugin.cfg" "${PLUGINS_DIR}/channelvol/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:ChannelVolPlugin>" "${PLUGINS_DIR}/channelvol"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:ChannelVolPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS AND PluginPlatform STREQUAL "windows-mingw")
      add_custom_command(TARGET ChannelVolPlugin POST_BUILD
         COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/channelvol/libgcc_s_seh-1.dll"
         COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libstdc++-6.dll" "${PLUGINS_DIR}/channelvol/libstdc++-6.dll"
      )
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(ChannelVolPlugin STATIC ${CHANNELVOL_PLUGIN_SOURCES})

   target_include_directories(ChannelVolPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/channelvol
   )

   set_target_properties(ChannelVolPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-channelvol"
   )
endif()
